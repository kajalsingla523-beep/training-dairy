We explored CallableStatement to work with stored procedures in MySQL. This allows Java programs to call predefined database routines for complex operations.

import java.sql.*;
class CallableExample {
    public static void main(String[] args) {
        try {
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/company", "root", "password");
            CallableStatement cs = con.prepareCall("{call getEmployee(?)}");
            cs.setInt(1, 101);
            ResultSet rs = cs.executeQuery();
            while(rs.next()) {
                System.out.println(rs.getString("name"));
            }
        } catch(Exception e) {
            e.printStackTrace();
        }
    }
}
